local function GenerateRandomInRange(min, max)
    math.randomseed(os.time())
    local r = math.random(min, max)
    return r
end

local function IsNoNumber(a)
    local b = tonumber(a)
    if not b then
        repeat
            print("–ù–µ —è–≤–ª—è–µ—Ç—Å—è —á–∏—Å–ª–æ–º. –í–≤–µ–¥–∏—Ç–µ —á–∏—Å–ª–æ")
            local c = io.read()
            a = IsNoNumber(c)
        until a ~= nil
        return a
    else
        return b
    end
end

        response = {
            answer = string.format("–í—ã –æ—Ç–≥–∞–¥–∞–ª–∏! –ó–∞–≥–∞–¥–∞–Ω–Ω–æ–µ —á–∏—Å–ª–æ: %d", a),
            isWinner = true
        }
    elseif c > 0 then
        response = {
            answer = string.format("–ó–∞–≥–∞–¥–∞–Ω–Ω–æ–µ —á–∏—Å–ª–æ –±–æ–ª—å—à–µ. –ü–æ–ø—ã—Ç–æ–∫ –æ—Å—Ç–∞–ª–æ—Å—å: %d", l - 1),
            isWinner = false,
            life = l - 1
        }
    elseif c < 0 then
        response = {
            answer = string.format("–ó–∞–≥–∞–¥–∞–Ω–Ω–æ–µ —á–∏—Å–ª–æ –º–µ–Ω—å—à–µ. –ü–æ–ø—ã—Ç–æ–∫ –æ—Å—Ç–∞–ª–æ—Å—å: %d", l - 1),

    return response.isWinner, response.answer, response.life -- –≤–æ–∑–≤—Ä–∞—â–∞—é 2 –∑–Ω–∞—á–µ–Ω–∏—è, –ø–µ—Ä–≤–æ–µ - –ø–æ–±–µ–¥–∞ –∏–ª–∏ –Ω–µ—Ç, –≤—Ç–æ—Ä–æ–µ - –æ—Ç–≤–µ—Ç –ø—Ä–æ–≥—Ä–∞–º–º—ã

        if lives == 0 then

    end
